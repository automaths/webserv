
http{

	server {
		listen 8079;

		root ./www/default       ;
		
		index index.html index.htm     ;

		cgi .php /usr/lib/cgi-bin/php .py /usr/bin/python3       ;

		server_name default;

		allow_method POST    ; 

		autoindex on    ;

		upload_pass /path/upload    ;

		client_buffer_size  200M     ;

		limit_upload   ;

		rewrite    /path/    permanent    ;
		rewrite    /path2/    redirection    ;

		location /Downloads {

			limit_upload    ;
		upload_pass /path/upload    ;
			allow_method POST GET        ;
			root ./www/default/bouhbouh    ;
			client_buffer_size 200M   ;
			error_page 404 /path/to/404    ;
			index index.html     ;
			autoindex on    ;
			cgi .php     /path/to/php     ;
			rewrite    /path/    permanent    ;
			rewrite    /path2/    redirection    ;
		}

		error_page  404     ./www/default/w404.html       ;
		error_page  403     /403.html       ;
	}
}
