Http scope rules: 

    - client_body_buffer_size
    - error_page
    - root
    - index
    - autoindex
    - cgi

Server scope rules:

    - listen
    - server_name
    - index
    - client_body_buffer_size
    - autoindex
    - root
    - allow_method
    - cgi
    - error_page

Location scope rules:

    - client_body_buffer_size
    - error_page
    - root
    - index
    - autoindex
    - allow_method
    - cgi



http{

	server {
		listen 8079;

		root ./www/default       ;
		
		index index.html index.htm;

		cgi .php /usr/lib/cgi-bin/php .py /usr/bin/python3       ;

		server_name default;

		location /Downloads {
			allow_method POST GET        ;
			root ./www/default/bouhbouh    ;
			client_buffer_size 200M   ;
			error_page 404 /path/to/404    ;
			index index.html     ;
			autoindex on    ;
			cgi .php     /path/to/php     ;

		}

		error_page  404     ./www/default/w404.html       ;
		error_page  403     /403.html       ;
	}

	server {
		listen 8080;

		root ./www/basic;

		index index.html index.htm;

		server_name basic;

		error_page  404     /404.html;
		error_page  403     /403.html;
	}
}
